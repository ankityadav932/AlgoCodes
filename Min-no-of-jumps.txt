// Function to return minimum number of jumps to end of array

class Solution{
  public:
    int minJumps(int arr[], int n)
    {
        int rangeStart ,jumps, rangeStop, maxVal, maxInd;

        jumps = 0;
        rangeStart = 1;
        rangeStop = arr[0];
        maxInd = 0;
        maxVal = arr[0];

        while(rangeStop < n-1)
        {    
            for (int j = rangeStart; j <= rangeStop; ++j)
            {
                if (arr[j] > maxVal)
                {
                    maxVal = arr[j];
                    maxInd = j;
                }
            }

            rangeStart = maxInd+1;
            rangeStop = maxInd+maxVal;
            maxVal = arr[rangeStart];
            jumps++;

            if (rangeStop >= n-1) jumps++;
        }

        return jumps;
    }
};
