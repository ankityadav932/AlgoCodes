#include <iostream>
#include <string>
#include <vector>

using namespace std;

void brainGame(vector<int>nums) {
    char player = 'A';
    bool move = false;
    
    for(int i = 0; i < nums.size(); i++)
    {
        for(int j = 2; j < nums[i]; j++)
        {
            if(nums[i] % j == 0)
            {
                move = true;
                player= 'B';
                nums[i] = j;
                break;
            }
        }
        
        if(move)
        {
            move = false;
            i = 0;
            continue;
        }
    }
    
    if(player == 'A')
        cout << 'B';
    else
        cout << 'A';

}

int main()
{   
    vector<int> arr = {14};
    brainGame(arr);
    return 0;
}
