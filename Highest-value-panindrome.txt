void solve(string pd, int n)
{
    int i = 0;
    int j = pd.length()-1;
    vector<int> ind;
    while(i < j && n > 0)
    {
        if(pd[i] > pd[j])
        {
            pd[j] = pd[i];
            ind.push_back(j);
            n--;
        }
        else if(pd[j] > pd[i])
        {
            pd[i] = pd[j];
            ind.push_back(i);
            n--;
        }

        i++;
        j--;
    }

    i = 0;
    j = pd.length()-1;
    while(i < j && n > 0)
    {
        for(int k : ind)
        {
            if(i == k || j == k)
            {
                pd[i] = '9';
                pd[j] = '9';
                n--;  
            }
            else if(n > 1)
            {
                pd[i] = '9';
                pd[j] = '9';
                n -= 2;
            }
        }
        i++;
        j--;
    }
    
    i = 0;
    j = pd.length()-1;
    while(i < j)
    {
        if(pd[i] != pd[j])
        {
            cout << -1 << endl;
            return;
        }
        i++;
        j--;
    }

    for(char i : pd)
        cout << i;
}

vector<int> strToArr(string str)
{
    string temp;
    vector<int> arr;

    for (int i = 0; i < str.length(); ++i)
    {
        if (str[i] != ' ')
        {
            temp += str[i];
        }
        else
        {
            int x = stoi(temp);
            arr.push_back(x);
            temp = "";
        }
    }

    int x = stoi(temp);
    arr.push_back(x);

    return arr;
}
