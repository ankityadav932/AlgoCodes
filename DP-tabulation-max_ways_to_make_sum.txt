#include <bits/stdc++.h>
using namespace std;

int getAllWays(int arr[], int n, int t)
{
    int dpCol = t+1;
    int dpRow = n;
    int dp[dpRow][dpCol] = {0};

    for (int i = 0; i < dpRow; ++i)
    {
        for (int j = 0; j < dpCol; ++j)
        {
            if (j == 0) continue;

            if (i == 0)
            {
                int num = arr[i];
                if (j%num == 0) 
                    dp[i][j] = 1;
                else 
                    dp[i][j] = 0;


                continue;
            } 
            else if (i > 0)
            {
                int num = arr[i];

                if (j > num)
                {
                    dp[i][j] = dp[i][j-num] + dp[i-1][j];    
                }
                else if (j == num)
                {
                    dp[i][j] = dp[i-1][j] + 1;
                }
                else dp[i][j] = dp[i-1][j];
            }
        }
    }

    return dp[n-1][t];
}


int main()
{
    int n = 3;
    int t = 12;
    int arr[n] = {3, 5, 6};
    int ans = getAllWays(arr, n, t);
    cout << ans;
    return 0;
}
