class Solution {
  public:
    int getMinDiff(int arr[], int n, int k) 
    {
        int minn, maxx, mid;
        sort(arr, arr+n);

        for (int i = 0; i < n; ++i)
        {
            cout << arr[i] << ',';
        }

        cout << endl;
        return 0;

        minn = arr[0] + k;
        maxx = arr[n-1] - k;

        int currMin, currMax;

        for (int i = 0; i < n; ++i)
        {
            currMin = arr[i]-k;
            currMax = arr[i]+k;

            if (currMin < 0)
            {
                maxx = max(maxx, currMax);
                arr[i] = currMax;
                continue;
            }

            int maxDiff = (currMax > maxx) ? currMax - maxx : 0; 
            int minDiff = (currMin < minn) ? minn - currMin : 0;

            if (minDiff > maxDiff)
            {
                maxx = max(maxx, currMax);
                arr[i] = currMax; 
            }
            else
            {
                minn = min(minn, currMin);
                arr[i] = currMin;
            }
        }   

        return abs(maxx-minn);
    }
};

    int n, k;

    for (int i = 0; i < n; i++) {

    }
    Solution ob;
    auto ans = ob.getMinDiff(arr, n, k);
